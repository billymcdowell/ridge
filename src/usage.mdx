# Usage Guide

Learn how to effectively use Ridge UI components in your applications with detailed examples, best practices, and advanced techniques.

## Component Overview

Ridge UI currently provides three core components, each designed with flexibility and accessibility in mind:

- **Button Component**: Interactive buttons with multiple variants and states
- **Input Component**: Form inputs with validation and customization options
- **Dropdown Component**: Select dropdowns with advanced features like multi-select and search

## Button Component

The button component is the foundation of user interactions in your application.

### Basic Usage

```html
<rdg-button>Default Button</rdg-button>
```

### Variants

Choose from five distinct button styles:

```html
<!-- Primary button for main actions -->
<rdg-button variant="primary">Primary Action</rdg-button>

<!-- Secondary button for supporting actions -->
<rdg-button variant="secondary">Secondary Action</rdg-button>

<!-- Outline button for subtle actions -->
<rdg-button variant="outline">Outline Button</rdg-button>

<!-- Ghost button for minimal presence -->
<rdg-button variant="ghost">Ghost Button</rdg-button>

<!-- Danger button for destructive actions -->
<rdg-button variant="danger">Delete Item</rdg-button>
```

### Sizes

Three sizes to fit different contexts:

```html
<rdg-button size="small">Small</rdg-button>
<rdg-button size="medium">Medium</rdg-button>
<rdg-button size="large">Large</rdg-button>
```

### States and Properties

```html
<!-- Disabled state -->
<rdg-button disabled>Disabled Button</rdg-button>

<!-- Loading state with spinner -->
<rdg-button loading>Loading...</rdg-button>

<!-- Link button (renders as anchor tag) -->
<rdg-button href="https://example.com">Visit Website</rdg-button>

<!-- Link button with target -->
<rdg-button href="https://example.com" target="_blank">Open in New Tab</rdg-button>
```

### Event Handling

```javascript
// Listen for button clicks
document.addEventListener('button-click', (event) => {
  console.log('Button clicked:', event.detail);
  console.log('Button element:', event.target);
});

// Or attach to specific button
const button = document.querySelector('#my-button');
button.addEventListener('button-click', (event) => {
  // Handle click
});
```

## Best Practices

### Accessibility

```html
<!-- Always provide meaningful labels -->
<rdg-input label="Email Address" type="email" required></rdg-input>

<!-- Use appropriate button text -->
<rdg-button variant="danger">Delete Account</rdg-button>

<!-- Provide error messages for validation -->
<rdg-input 
  label="Password" 
  type="password" 
  required
  error-message="Password must be at least 8 characters">
</rdg-input>
```

### Performance

```javascript
// Import only what you need (when tree-shaking is available)
import { RdgButton } from 'ridge-ui/button';

// Use event delegation for multiple components
document.addEventListener('button-click', (event) => {
  if (event.target.matches('.action-button')) {
    // Handle action buttons
  }
});
```

### Form Validation

```html
<form id="registration-form">
  <rdg-input 
    name="email"
    type="email" 
    label="Email" 
    required
    pattern="[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$"
    error-message="Please enter a valid email address">
  </rdg-input>
  
  <rdg-input 
    name="password"
    type="password" 
    label="Password" 
    required
    minlength="8"
    error-message="Password must be at least 8 characters">
  </rdg-input>
  
  <rdg-button variant="primary" type="submit">Register</rdg-button>
</form>

<script>
// Custom validation
document.getElementById('registration-form').addEventListener('submit', (e) => {
  e.preventDefault();
  
  const email = e.target.querySelector('[name="email"]');
  const password = e.target.querySelector('[name="password"]');
  
  // Custom validation logic
  if (!email.checkValidity()) {
    email.setAttribute('error-message', 'Please enter a valid email');
    return;
  }
  
  if (!password.checkValidity()) {
    password.setAttribute('error-message', 'Password is too short');
    return;
  }
  
  // Form is valid, submit data
  console.log('Form submitted successfully');
});
</script>
```

## Advanced Usage Patterns

### Dynamic Component Creation

```javascript
// Create components programmatically
function createButton(text, variant = 'primary') {
  const button = document.createElement('rdg-button');
  button.variant = variant;
  button.textContent = text;
  
  button.addEventListener('button-click', () => {
    console.log(`${text} clicked`);
  });
  
  return button;
}

// Usage
const saveButton = createButton('Save', 'primary');
const cancelButton = createButton('Cancel', 'secondary');

document.body.appendChild(saveButton);
document.body.appendChild(cancelButton);
```

### Component State Management

```javascript
class FormManager {
  constructor(formElement) {
    this.form = formElement;
    this.inputs = formElement.querySelectorAll('rdg-input');
    this.submitButton = formElement.querySelector('rdg-button[type="submit"]');
    
    this.setupValidation();
  }
  
  setupValidation() {
    this.inputs.forEach(input => {
      input.addEventListener('input-change', () => {
        this.validateForm();
      });
    });
  }
  
  validateForm() {
    const isValid = Array.from(this.inputs).every(input => 
      input.checkValidity()
    );
    
    this.submitButton.disabled = !isValid;
  }
  
  getFormData() {
    const data = {};
    this.inputs.forEach(input => {
      data[input.name] = input.value;
    });
    return data;
  }
}

// Usage
const formManager = new FormManager(document.getElementById('my-form'));
```

### Custom Event Handling

```javascript
// Create a custom component wrapper
class RidgeFormField extends HTMLElement {
  constructor() {
    super();
    this.innerHTML = `
      <rdg-input></rdg-input>
      <div class="field-help"></div>
    `;
    
    this.input = this.querySelector('rdg-input');
    this.help = this.querySelector('.field-help');
    
    this.setupEventHandlers();
  }
  
  setupEventHandlers() {
    this.input.addEventListener('input-change', (e) => {
      this.dispatchEvent(new CustomEvent('field-change', {
        detail: { value: e.detail.value, field: this }
      }));
    });
  }
  
  set label(value) {
    this.input.setAttribute('label', value);
  }
  
  set helpText(value) {
    this.help.textContent = value;
  }
}

customElements.define('ridge-form-field', RidgeFormField);
```
